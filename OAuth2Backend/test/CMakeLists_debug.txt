cmake_minimum_required(VERSION 3.20)
project(OAuth2Test_debug)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find dependencies
find_package(Drogon CONFIG REQUIRED)
find_package(OpenSSL REQUIRED)
find_package(pthread REQUIRED)

# Debug executable
add_executable(OAuth2Test_test_debug
    test_main_debug.cc
    ${TEST_SOURCES}
)

# Link libraries
target_link_libraries(OAuth2Test_test_debug
    PRIVATE
    Drogon::Drogon
    ${OPENSSL_LIBRARIES}
    Threads::Threads
)

# Include directories
target_include_directories(OAuth2Test_test_debug
    PRIVATE
    ${CMAKE_SOURCE_DIR}
    ${CMAKE_SOURCE_DIR}/..
)

# Compiler flags for debugging
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    target_compile_options(OAuth2Test_test_debug
        PRIVATE
        -Wall
        -Wextra
        -g
        -fsanitize=address
        -fno-omit-frame-pointer
    )
    target_link_options(OAuth2Test_test_debug
        PRIVATE
        -fsanitize=address
    )
endif()
